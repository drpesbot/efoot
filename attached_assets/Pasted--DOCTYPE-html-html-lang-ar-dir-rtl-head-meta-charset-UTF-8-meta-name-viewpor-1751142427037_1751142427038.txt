        // رفع الصور
        function setupDragDrop() {
            const dropArea = document.querySelector('.drag-drop-area');
            
            ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
                dropArea.addEventListener(eventName, preventDefaults, false);
            });

            function preventDefaults(e) {
                e.preventDefault();
                e.stopPropagation();
            }

            ['dragenter', 'dragover'].forEach(eventName => {
                dropArea.addEventListener(eventName, highlight, false);
            });

            ['dragleave', 'drop'].forEach(eventName => {
                dropArea.addEventListener(eventName, unhighlight, false);
            });

            function highlight(e) {
                dropArea.classList.add('dragover');
            }

            function unhighlight(e) {
                dropArea.classList.remove('dragover');
            }

            dropArea.addEventListener('drop', handleDrop, false);

            function handleDrop(e) {
                const dt = e.dataTransfer;
                const files = dt.files;
                handleFiles(files);
            }

            function handleFiles(files) {
                if (files.length > 0) {
                    const file = files[0];
                    if (file.type.startsWith('image/')) {
                        const reader = new FileReader();
                        reader.onload = function(e) {
                            handleImageUpload({ target: { files: [file] } });
                        };
                        reader.readAsDataURL(file);
                    }
                }
            }
        }

        function handleImageUpload(event) {
            const file = event.target.files[0];
            if (file && file.type.startsWith('image/')) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    currentPlayerImage = e.target.result;
                    
                    const preview = document.getElementById('imagePreview');
                    preview.innerHTML = `
                        <img src="${currentPlayerImage}" alt="صورة اللاعب" class="w-24 h-24 rounded-full mx-auto object-cover border-2 border-blue-400">
                        <p class="mt-2 text-green-400">تم رفع الصورة بنجاح</p>
                    `;
                };
                reader.readAsDataURL(file);
            }
        }

        // إضافة لاعب
        function addPlayer() {
            const name = document.getElementById('playerName').value.trim();
            const rating = parseInt(document.getElementById('playerRating').value);
            const position = document.getElementById('playerPosition').value;
            const team = document.getElementById('playerTeam').value.trim();
            const nationality = document.getElementById('playerNationality').value.trim();

            // التحقق من صحة البيانات
            if (!name || !rating || !position || !team || !nationality) {
                showNotification('يرجى ملء جميع الحقول المطلوبة', 'error');
                return;
            }

            if (rating < 50 || rating > 99) {
                showNotification('التقييم يجب أن يكون بين 50 و 99', 'error');
                return;
            }

            // التحقق من عدم وجود لاعب بنفس الاسم
            if (players.some(player => player.name.toLowerCase() === name.toLowerCase())) {
                showNotification('يوجد لاعب بهذا الاسم مسبقاً', 'error');
                return;
            }

            const newPlayer = {
                id: nextPlayerId++,
                name: name,
                rating: rating,
                position: position,
                team: team,
                nationality: nationality,
                image: currentPlayerImage,
                stats: {
                    shooting: parseInt(document.getElementById('shooting').value) || 70,
                    passing: parseInt(document.getElementById('passing').value) || 70,
                    dribbling: parseInt(document.getElementById('dribbling').value) || 70,
                    dexterity: parseInt(document.getElementById('dexterity').value) || 70,
                    lowerBodyStrength: parseInt(document.getElementById('lowerBodyStrength').value) || 70,
                    aerialStrength: parseInt(document.getElementById('aerialStrength').value) || 70,
                    defending: parseInt(document.getElementById('defending').value) || 70,
                    gk1: parseInt(document.getElementById('gk1').value) || 70,
                    gk2: parseInt(document.getElementById('gk2').value) || 70,
                    gk3: parseInt(document.getElementById('gk3').value) || 70
                }
            };

            players.push(newPlayer);
            saveData();
            displayPlayers();
            closeModal('addPlayerModal');
            showNotification(تم إضافة اللاعب ${name} بنجاح, 'success');
        }

        // إعادة تعيين نموذج إضافة لاعب
        function resetAddPlayerForm() {
            document.getElementById('playerName').value = '';
            document.getElementById('playerRating').value = '';
            document.getElementById('playerPosition').value = '';
            document.getElementById('playerTeam').value = '';
            document.getElementById('playerNationality').value = '';
            
            // إعادة تعيين الإحصائيات
            ['shooting', 'passing', 'dribbling', 'dexterity', 'lowerBodyStrength', 'aerialStrength', 'defending', 'gk1', 'gk2', 'gk3'].forEach(stat => {
                document.getElementById(stat).value = 70;
            });
            
            // إعادة تعيين الصورة
            currentPlayerImage = null;
            document.getElementById('imagePreview').innerHTML = `
                <i class="fas fa-camera text-4xl text-blue-400 mb-2"></i>
                <p>اضغط لرفع صورة أو اسحب الصورة هنا</p>
            `;
            document.getElementById('playerImageInput').value = '';
        }

        // حذف لاعب من الجدول
        function deletePlayerById(playerId) {
            if (confirm('هل أنت متأكد من حذف هذا اللاعب؟')) {
                players = players.filter(player => player.id !== playerId);
                saveData();
                displayPlayers();
                showNotification('تم حذف اللاعب بنجاح', 'success');
            }
        }

        // حذف لاعب من النموذج
        function deletePlayer() {
            const playerId = parseInt(document.getElementById('deletePlayerSelect').value);
            
            if (!playerId) {
                showNotification('يرجى اختيار لاعب للحذف', 'error');
                return;
            }

            const player = players.find(p => p.id === playerId);
            if (!player) {
                showNotification('اللاعب غير موجود', 'error');
                return;
            }

            if (confirm(هل أنت متأكد من حذف اللاعب ${player.name}؟)) {
                players = players.filter(p => p.id !== playerId);
                saveData();
                displayPlayers();
                closeModal('deletePlayerModal');
                showNotification(تم حذف اللاعب ${player.name} بنجاح, 'success');
            }
        }

        // تصدير البيانات
        function exportData() {
            try {
                const dataStr = JSON.stringify(players, null, 2);
                const dataBlob = new Blob([dataStr], {type: 'application/json'});
                
                const link = document.createElement('a');
                link.href = URL.createObjectURL(dataBlob);
                link.download = efootball_players_${new Date().toISOString().split('T')[0]}.json;
                link.click();
                
                showNotification('تم تصدير البيانات بنجاح', 'success');
            } catch (error) {
                showNotification('خطأ في تصدير البيانات', 'error');
            }
        }
